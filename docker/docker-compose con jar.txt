services:
  # 🐘 Base de datos PostgreSQL
  db:
    image: postgres:18-alpine                # Imagen oficial de Postgres
    container_name: prodmind_db              # Nombre del contenedor
    environment:
      POSTGRES_USER: prodmind                # Usuario del PostgreSQL
      POSTGRES_PASSWORD: prodmind            # Contraseña del PostgreSQL
      POSTGRES_DB: prodmind                  # Base de datos por defecto
    ports:
      - "5432:5432"                          # Puerto expuesto (host:contenedor)
    volumes:
      - db_data:/var/lib/postgresql/data     # Volumen persistente de datos
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U prodmind"]  # Comando para verificar si DB está lista
      interval: 5s
      timeout: 5s
      retries: 5
    networks:
      - prodmind_net
    restart: unless-stopped                  # Reinicia si el contenedor falla

  # ☕ API REST (Spring Boot)
  backend:
    build: /Users/davididarragagiraldo/Library/CloudStorage/OneDrive-Personal/Personal/Proyectos de software/focoprod/backend                      # Carpeta donde está tu Dockerfile
    container_name: prodmind_backend
    depends_on:
      db:
        condition: service_healthy            # Espera que PostgreSQL esté listo antes de arrancar
    environment:
      SPRING_DATASOURCE_URL: jdbc:postgresql://db:5432/prodmind
      SPRING_DATASOURCE_USERNAME: prodmind
      SPRING_DATASOURCE_PASSWORD: prodmind
      SPRING_JPA_HIBERNATE_DDL_AUTO: update
      SPRING_JPA_SHOW_SQL: "true"
      SPRING_PROFILES_ACTIVE: docker
    ports:
      - "8080:8080"                          # Puerto expuesto del backend
    networks:
      - prodmind_net
    restart: unless-stopped

# 🌐 Red compartida entre backend y base de datos
